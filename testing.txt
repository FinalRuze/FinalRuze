Set objWMIService = GetObject("winmgmts:\\.\root\cimv2")
Set objNewUser = objWMIService.Get("Win32_UserAccount").SpawnInstance_

strUsername = "NewUser"
strPassword = "NewUserPassword"
strDescription = "New User Account"
intFlags = 512 ' Normal user account type

objNewUser.Name = strUsername
objNewUser.Password = strPassword
objNewUser.Description = strDescription
objNewUser.AccountType = intFlags

objWMIService.Get("Win32_UserAccount").Create objNewUser, intReturnValue

If intReturnValue = 0 Then
    WScript.Echo "User account created successfully!"
Else
    WScript.Echo "Error creating user account."
End If
