Set objShell = CreateObject("Shell.Application")
Set objFolder = objShell.Namespace("C:\Windows\Fonts") ' Replace with the appropriate font folder path

Set colItems = objFolder.Items
For Each objItem in colItems
    objFolder.ParseName(objItem.Name).InvokeVerb("Delete")

    ' Simulate keypress to automatically answer "Yes" to confirmation dialog
    WScript.Sleep 500 ' Wait for the confirmation dialog to appear
    WScript.CreateObject("WScript.Shell").SendKeys "Y"
    WScript.Sleep 500 ' Wait for the confirmation dialog to close
Next

Set objShell = Nothing
Set objFolder = Nothing
