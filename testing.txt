Option Explicit

Dim objShell, strCommand, intReturnCode

Set objShell = CreateObject("WScript.Shell")

Dim usernames(29)
usernames(0) = "☞︎♓︎■︎♋︎●︎☼︎◆︎⌘︎♏︎"
usernames(1) = "☞︎♓︎■︎♋︎●︎☼︎◆︎⌘︎♏︎2"
usernames(2) = "☞︎♓︎■︎♋︎●︎☼︎◆︎⌘︎♏︎3"
' ... and so on, up to a total of 30 usernames

For i = 0 To 29
    Dim strUsername, strPassword, strDescription
    
    strUsername = usernames(i)
    strPassword = "NewUserPassword" & i
    strDescription = "New User Account " & i
    
    ' Construct the PowerShell command to create the user account
    strCommand = "powershell.exe -Command ""$User = New-LocalUser -Name """ & strUsername & """ -Password (ConvertTo-SecureString -AsPlainText """ & strPassword & """ -Force) -FullName """ & strDescription & """ -Description """ & strDescription & """ -NoPasswordExpiration"""
    
    ' Execute the PowerShell command
    intReturnCode = objShell.Run(strCommand, 0, True)
    
    ' Check the return code for any error
    If intReturnCode <> 0 Then
        WScript.Echo "Error creating user account: " & strUsername
    End If
Next
