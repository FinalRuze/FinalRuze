using System;
using System.Drawing;
using System.Windows.Forms;

public class GdiEffectsForm : Form
{
    private Timer timer;
    private int effectIndex;

    public GdiEffectsForm()
    {
        this.FormBorderStyle = FormBorderStyle.None;
        this.BackColor = Color.Black;
        this.Bounds = Screen.PrimaryScreen.Bounds;

        this.timer = new Timer();
        this.timer.Interval = 3000; // Change interval (in milliseconds) to adjust effect speed
        this.timer.Tick += Timer_Tick;
        this.timer.Start();

        this.effectIndex = 0;
    }

    protected override void OnPaint(PaintEventArgs e)
    {
        base.OnPaint(e);
        Graphics g = e.Graphics;

        switch (effectIndex)
        {
            case 0:
                // Effect 1: Gradient background
                using (LinearGradientBrush brush = new LinearGradientBrush(
                    ClientRectangle,
                    Color.Red,
                    Color.Blue,
                    LinearGradientMode.Vertical))
                {
                    g.FillRectangle(brush, ClientRectangle);
                }
                break;

            case 1:
                // Effect 2: Random colored lines
                Random random = new Random();
                for (int i = 0; i < 1000; i++)
                {
                    using (Pen pen = new Pen(Color.FromArgb(random.Next(256), random.Next(256), random.Next(256))))
                    {
                        g.DrawLine(pen, random.Next(ClientSize.Width), random.Next(ClientSize.Height),
                                   random.Next(ClientSize.Width), random.Next(ClientSize.Height));
                    }
                }
                break;

            // Add more effects here...

            default:
                this.timer.Stop();
                this.Close();
                break;
        }
    }

    private void Timer_Tick(object sender, EventArgs e)
    {
        effectIndex++;
        Refresh();
    }
}

public static class Program
{
    public static void Main()
    {
        Application.Run(new GdiEffectsForm());
    }
}
