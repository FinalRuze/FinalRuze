Option Explicit

Dim MorseCodeDict
Set MorseCodeDict = CreateObject("Scripting.Dictionary")
MorseCodeDict.Add "A", ".-"
MorseCodeDict.Add "B", "-..."
MorseCodeDict.Add "C", "-.-."
MorseCodeDict.Add "D", "-.."
MorseCodeDict.Add "E", "."
' ... Add more mappings for the remaining letters

Dim message
message = "HELLO" ' The message you want to convert to Morse code

Dim i, j, morseCodeMessage
morseCodeMessage = ""
For i = 1 To Len(message)
    Dim letter, morseCode
    letter = UCase(Mid(message, i, 1))
    
    If MorseCodeDict.Exists(letter) Then
        morseCode = MorseCodeDict(letter)
        
        For j = 1 To Len(morseCode)
            Dim symbol
            symbol = Mid(morseCode, j, 1)
            
            If symbol = "." Then
                morseCodeMessage = morseCodeMessage & "."
            ElseIf symbol = "-" Then
                morseCodeMessage = morseCodeMessage & "-"
            End If
            
            WScript.Sleep 100 ' Pause between symbols
        Next
        
        morseCodeMessage = morseCodeMessage & " " ' Add a space between letters
        WScript.Sleep 200 ' Pause between letters
    End If
Next

MsgBox morseCodeMessage, vbInformation, "Morse Code"
